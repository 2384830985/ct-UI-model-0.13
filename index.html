tree使用及展示：

icon图标链接
 
        
              var ChituTree = [                //数组格式(必须这样) 后面会直接转换成各自的层级
              {
                  id: 1,
                  name: "一级标题",
                  pid: 0
              }, {
                  id: 2,
                  name: "二级标题",
                  pid: 0
              }]
              ChiTu.CtTree({
                iconKai: 'icon-kaiwenjianjia',//可设置开启时样式默认'icon-kaiwenjianjia'
                iconGuan: 'icon-wenjianjia',  //可设置关闭时样式默认'icon-wenjianjia'
                dingDan: 'icon-dingdan',      //可设置文件样式默认'icon-dingdan'
                guanBi: 'icon-guanbi1',       //关闭按钮样式'icon-guanbi1'
                tree: '#tree',                //插入的dom节点必须
                arr: ChituTree,               //插入的数组必须
                ok: function (res) {          //点击标题回调函数
                    console.log(res.id);
                    console.log(res.name);
                }
              })
